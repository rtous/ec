\begin{document}

\begin{tikzpicture}[
    node distance=1.5cm and 2cm,
    address box/.style={draw, rectangle, minimum height=1cm, minimum width=4cm, align=center, font=\large},
    pt entry/.style={draw, rectangle, minimum height=0.6cm, minimum width=0.8cm, align=center, font=\small},
    label font/.style={font=\small},
    bit label/.style={font=\tiny, text height=1.2ex, text depth=0.2ex},
    >={Stealth[scale=1]}
]

% Logical Address
\node[address box, minimum width=8cm] (logical_addr) {VPN \hspace{2cm} page offset};
\node[label font, left=0.1cm of logical_addr] {\textbf{Adreça lògica}};

% Logical Address Bits (Top)
\foreach \bit/\label [count=\i] in {31/31, 30/30, ..., 0/0} {
    \pgfmathsetmacro{\xpos}{(\i-1)*8/31}
    \node[bit label, anchor=base, text width=0.1cm, align=center] at ($(logical_addr.north west) + (\xpos, 0.1)$) {\bit};
}

% Bit Labels Below Logical Address
\draw[|-|] ($(logical_addr.north west) + (0, 0.3)$) -- ($(logical_addr.north west) + (5, 0.3)$) node[midway, above] {20 bits};
\draw[|-|] ($(logical_addr.north west) + (5, 0.3)$) -- ($(logical_addr.north west) + (8, 0.3)$) node[midway, above] {12 bits};
\draw (5, 0.5) -- (5, -0.5); % Internal divider

% Page Table Register and Label
\node[draw, rectangle, text width=4cm, align=center, below left=1.2cm and 1cm of logical_addr] (pt_reg) {Registre de Taula de Pàgines};
\draw[->] (pt_reg) -- ($(pt_reg.east) + (0.8,0)$);

% Taula de Pàgines Label
\node[font=\Large\bfseries, below=1cm of pt_reg] (tp_label) {Taula de Pàgines};

% Page Table Grid
\begin{scope}[shift={($(logical_addr.south west) + (2,-2.5)$)}]
    \node[anchor=south west, label font] at (0, 0.2) {VPN};
    \node[anchor=south west, label font] at (0.8, 0.2) {P};
    \node[anchor=south west, label font] at (1.6, 0.2) {D};
    \node[anchor=south west, label font] at (2.4, 0.2) {PPN};
    
    \foreach \y [count=\j] in {0, 1, 2, 3, 4, 5} {
        \node[pt entry, minimum width=0.8cm] at (0, -\y*0.6) (v\j) {\ifnum\j=1 0\fi \ifnum\j=2 1\fi \ifnum\j=3 ...\fi \ifnum\j=4 ...\fi \ifnum\j=5 ...\fi};
        \node[pt entry, minimum width=0.8cm] at (0.8, -\y*0.6) (p\j) {};
        \node[pt entry, minimum width=0.8cm] at (1.6, -\y*0.6) (d\j) {};
        \node[pt entry, minimum width=2.4cm] at (3.2, -\y*0.6) (ppn\j) {};
    }
    % Highlighted Row
    \node[pt entry, fill=gray!30, minimum width=0.8cm] at (0.8, -3*0.6) {};
    \node[pt entry, fill=gray!30, minimum width=0.8cm] at (1.6, -3*0.6) {};
    \node[pt entry, fill=gray!30, minimum width=2.4cm] at (3.2, -3*0.6) {};
    \fill (0.8, -3*0.6) circle (2pt);
    \fill (3.2, -3*0.6) circle (2pt);

    \node[anchor=north east, font=\normalsize] at (-0.2, 0.1) {0};
    \node[anchor=north east, font=\normalsize] at (-0.2, -0.5) {1};
    \node[anchor=north east, font=\normalsize] at (-0.2, -1.1) {...};
    \node[anchor=north east, font=\normalsize] at (-0.2, -2.1) {...};
    \node[anchor=north east, font=\normalsize] at (-0.1, -2.8) {$2^{20}-1$};

    \draw[dashed] (0.4, 1.2*0.6) -- (0.4, 1.8*0.6);
    \draw[dashed] (1.2, 1.2*0.6) -- (1.2, 1.8*0.6);
    \node at (2.0, 1.5*0.6) {...};
    \node at (3.6, 1.5*0.6) {...};
    
    \draw[dashed] (0.4, -4.2*0.6) -- (0.4, -4.8*0.6);
    \draw[dashed] (1.2, -4.2*0.6) -- (1.2, -4.8*0.6);
    \node at (2.0, -4.5*0.6) {...};
    \node at (3.6, -4.5*0.6) {...};

\end{scope}

% Physical Address
\begin{scope}[shift={($(logical_addr.south west) + (4,-7)$)}]
    \node[address box, minimum width=5.2cm] (phys_addr) {PPN \hspace{1cm} page offset};
    \node[label font, left=0.1cm of phys_addr] {\textbf{Adreça física}};
    
    % Physical Address Bits (Bottom)
    \foreach \bit/\label [count=\i] in {13/13, 12/12, ..., 0/0} {
        \pgfmathsetmacro{\xpos}{(\i-1)*5.2/13}
        \node[bit label, anchor=base, text width=0.1cm, align=center] at ($(phys_addr.south west) + (\xpos, -0.2)$) {\bit};
    }

    % Bit Labels Above Physical Address
    \draw[|-|] ($(phys_addr.south west) + (0, -0.4)$) -- ($(phys_addr.south west) + (1.2, -0.4)$) node[midway, below] {2 bits};

    \draw (1.2, -0.5) -- (1.2, 0.5); % Internal divider
\end{scope}

% Connections
% VPN -> Table Row
\draw[->] ($(logical_addr.south west) + (2.5,0)$) -- ($(logical_addr.south west) + (2.5,-3.3)$) -- ($(logical_addr.south west) + (3.3,-3.3)$);

\draw[->] ($(logical_addr.south west) + (4.1,-3.3)$) -- ($(logical_addr.south west) + (4.1,-5.5)$) -- ($(logical_addr.south west) + (3.0,-5.5)$);
\node[text width=4.5cm, align=right, left=0.1cm of $(logical_addr.south west) + (3.0,-5.5)$] (note) {
Bit de presència (P). Si és 0 la pàgina està al disc. Si és 1 està a la memòria
};
